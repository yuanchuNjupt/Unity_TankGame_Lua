




Object:subClass("UIManager")

--依赖
local resourcesManager = require("Manager/ResourcesManager")

--成员属性
UIManager.panelDic = nil
UIManager.canvas = nil;


--构造函数
function UIManager:__init()
    self.panelDic = {}
    self.canvas = resourcesManager.LoadInstance("UI/UIRoot")
    GameObject.DontDestroyOnLoad(self.canvas)
end
--成员方法
function UIManager:ShowPanel(panelName)
    if self.panelDic[panelName] == nil then
        --第一次Show，从Resources加载面板
        local panelPrefab = resourcesManager.LoadInstance("UI/" .. panelName)
        if panelPrefab ~= nil then
            panelPrefab.transform:SetParent(self.canvas.transform, false) 
            -- 使用:new()实例化面板类，这样才会调用__init方法
            local panelInstance = _G[panelName]:new(panelPrefab)
            self.panelDic[panelName] = panelInstance
            print("实例化面板: " .. panelName)
        else
            print("未找到面板: " .. panelName)
        end
    else
        self.panelDic[panelName].panelObject:SetActive(true)
    end
end

function UIManager:HidePanel(panelName)
    if self.panelDic[panelName] == nil then
        print("面板未找到: " .. panelName)
        return
    end
    
    self.panelDic[panelName].panelObject:SetActive(false)
end

--全局唯一实例
local instance = UIManager:new()
return instance
