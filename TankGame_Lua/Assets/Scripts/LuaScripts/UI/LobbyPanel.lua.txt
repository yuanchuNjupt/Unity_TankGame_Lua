
BasePanel:subClass("LobbyPanel")

function LobbyPanel:__init(panelObject)
    -- 调用父类的Init方法
    self:Init(panelObject)
    
    -- 查找EntryBattle按钮
    local EntryBattleButton = self:FindComponentByName("EntryBattle", Button)
    
    if EntryBattleButton ~= nil then
        EntryBattleButton.onClick:AddListener(function()
            print("点击了进入战斗按钮")
        end)
    end
end
